project-root/
│
├── backend/
│   ├── config/           # Environment configs, database connection, app settings
│   ├── controllers/      # Functions handling request logic for routes
│   ├── models/           # Database schema models (Mongoose for MongoDB)
│   ├── routes/           # API endpoints definitions
│   ├── utils/            # Helper functions, e.g., hashing, formatting
│   └── middleware/       # Middleware like auth, logging, error handling
│
├── frontend/
│   ├── public/           # Static assets served directly (index.html, images, fonts)
│   │   └── index.html    # Root HTML file
│   │
│   └── src/
│       ├── components/       # Reusable React components
│       │   ├── common/       # Shared components like buttons, inputs
│       │   ├── auth/         # Login-related components
│       │   ├── dashboard/    # Components for dashboard page
│       │   ├── ec/           # Components for Eligible Couple pages
│       │   ├── pw/           # Components for Pregnant Women pages
│       │   └── child/        # Components for Child pages
│       │
│       ├── pages/            # Full page components (mapped to routes)
│       ├── services/         # API calls and data fetching logic
│       ├── styles/           # Global and page-specific CSS/SCSS
│       ├── context/          # React Context for global state management
│       └── utils/            # Frontend helper functions
│
└── database/
    └── scripts/             # SQL scripts for creating tables, seed data, etc.
